67  /**
68   * A {@link Service} which is defined by a {@link Path} or HTTP method annotations.
69   * This class is not supposed to be instantiated by a user. Please check out the documentation
70   * <a href="https://line.github.io/armeria/server-annotated-service.html#annotated-http-service">
71   * Annotated HTTP Service</a> to use this.
72   */
73  public class AnnotatedHttpService implements HttpService {
74      private static final Logger logger = LoggerFactory.getLogger(AnnotatedHttpService.class);
75  
76      static final ServiceLoader<ResponseConverterFunctionProvider> responseConverterFunctionProviders =
77              ServiceLoader.load(ResponseConverterFunctionProvider.class,
78                                 AnnotatedHttpService.class.getClassLoader());
79  
80      private final Object object;
81      private final Method method;
82      private final List<AnnotatedValueResolver> resolvers;
83  
84      private final AggregationStrategy aggregationStrategy;
85      private final List<ExceptionHandlerFunction> exceptionHandlers;
86      private final List<ResponseConverterFunction> responseConverters;
87  
88      @Nullable
89      private final ResponseConverterFunction providedResponseConverter;
90      private final PathMapping pathMapping;
91  
92      private final ResponseType responseType;
93  
94      AnnotatedHttpService(Object object, Method method,
95                           List<AnnotatedValueResolver> resolvers,
96                           List<ExceptionHandlerFunction> exceptionHandlers,
97                           List<ResponseConverterFunction> responseConverters,
98                           PathMapping pathMapping) {
99          this.object = requireNonNull(object, "object");
100         this.method = requireNonNull(method, "method");
101         this.resolvers = requireNonNull(resolvers, "resolvers");
102         this.exceptionHandlers = ImmutableList.copyOf(
103                 requireNonNull(exceptionHandlers, "exceptionHandlers"));
104         this.responseConverters = ImmutableList.copyOf(
105                 requireNonNull(responseConverters, "responseConverters"));
106 
107         aggregationStrategy = AggregationStrategy.from(resolvers);
108         providedResponseConverter = fromProvider(method);
109         this.pathMapping = requireNonNull(pathMapping, "pathMapping");
110 
111         final Class<?> returnType = method.getReturnType();
112         if (providedResponseConverter != null) {
113             responseType = ResponseType.HANDLED_BY_SPI;
114         } else if (HttpResponse.class.isAssignableFrom(returnType)) {
115             responseType = ResponseType.HTTP_RESPONSE;
116         } else if (CompletionStage.class.isAssignableFrom(returnType)) {
117             responseType = ResponseType.COMPLETION_STAGE;
118         } else {
119             responseType = ResponseType.OTHER_OBJECTS;
120         }
121 
122         this.method.setAccessible(true);
123     }
124 
125     @Nullable
126     private ResponseConverterFunction fromProvider(Method method) {
127         final Type returnType = method.getGenericReturnType();
128 
129         if (returnType instanceof ParameterizedType) {
130             final ParameterizedType p = (ParameterizedType) returnType;
131             if (Publisher.class.isAssignableFrom(toClass(p.getRawType())) &&
132                 Publisher.class.isAssignableFrom(toClass(p.getActualTypeArguments()[0]))) {
133                 throw new IllegalStateException(
134                         "Invalid return type of method '" + method.getName() + "'. " +
135                         "Cannot support '" + p.getActualTypeArguments()[0].getTypeName() +
136                         "' as a generic type of " + Publisher.class.getSimpleName());
137             }
138         }
139 
140         for (ResponseConverterFunctionProvider provider : responseConverterFunctionProviders) {
141             final ResponseConverterFunction func =
142                     provider.createResponseConverterFunction(returnType,
143                                                              this::convertResponse,
144                                                              this::convertException);
145             if (func != null) {
146                 return func;
147             }
148         }
149         return null;
150     }
151 
152     private static Class<?> toClass(Type type) {
153         if (type instanceof Class) {
154             return (Class<?>) type;
155         }
156         if (type instanceof ParameterizedType) {
157             return (Class<?>) ((ParameterizedType) type).getRawType();
158         }
159         return Void.class;
160     }

178     @Override
179     public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) throws Exception {
180         return HttpResponse.from(serve0(ctx, req));
181     }
182 
183     /**
184      * Executes the service method in different ways regarding its return type and whether the request is
185      * required to be aggregated. If the return type of the method is not a {@link CompletionStage} or
186      * {@link HttpResponse}, it will be executed in the blocking task executor.
187      */
188     private CompletionStage<HttpResponse> serve0(ServiceRequestContext ctx, HttpRequest req) {
189         final CompletableFuture<AggregatedHttpMessage> f =
190                 aggregationRequired(aggregationStrategy, req) ? req.aggregate()
191                                                               : CompletableFuture.completedFuture(null);
192         switch (responseType) {
193             case HANDLED_BY_SPI:
194                 return f.thenApply(msg -> {
195                     try {
196                         final Object obj = invoke(ctx, req, msg);
197                         if (obj instanceof HttpResponse) {
198                             return (HttpResponse) obj;
199                         }
200                         assert providedResponseConverter != null;
201                         return new ExceptionFilteredHttpResponse(
202                                 ctx, req, providedResponseConverter.convertResponse(ctx, obj));
203                     } catch (Throwable cause) {
204                         return convertException(ctx, req, cause);
205                     }
206                 });
207             case HTTP_RESPONSE:
208                 return f.thenApply(
209                         msg -> new ExceptionFilteredHttpResponse(ctx, req,
210                                                                  (HttpResponse) invoke(ctx, req, msg)));
211             case COMPLETION_STAGE:
212                 return f.thenCompose(msg -> toCompletionStage(invoke(ctx, req, msg)))
213                         .handle((result, cause) -> cause == null ? convertResponse(ctx, req, result)
214                                                                  : convertException(ctx, req, cause));
215             default:
216                 return f.thenApplyAsync(msg -> convertResponse(ctx, req, invoke(ctx, req, msg)),
217                                         ctx.blockingTaskExecutor());
218         }
219     }
220 
221     /**
222      * Invokes the service method with arguments.
223      */
224     private Object invoke(ServiceRequestContext ctx, HttpRequest req, @Nullable AggregatedHttpMessage message) {
225         try (SafeCloseable ignored = ctx.push(false)) {
226             final ResolverContext resolverContext = new ResolverContext(ctx, req, message);
227             final Object[] arguments = toArguments(resolvers, resolverContext);
228             return method.invoke(object, arguments);
229         } catch (Throwable cause) {
230             return convertException(ctx, req, cause);
231         }
232     }

234     /**
235      * Converts the specified {@code result} to {@link HttpResponse}.
236      */
237     private HttpResponse convertResponse(ServiceRequestContext ctx, HttpRequest req,
238                                          @Nullable Object result) {
239         if (result instanceof HttpResponse) {
240             return (HttpResponse) result;
241         }
242         if (result instanceof AggregatedHttpMessage) {
243             return HttpResponse.of((AggregatedHttpMessage) result);
244         }
245         if (result instanceof Publisher) {
246             final CompletableFuture<HttpResponse> future = new CompletableFuture<>();
247             final Publisher<?> publisher = (Publisher<?>) result;
248             publisher.subscribe(new PublisherToHttpResponseConverter(ctx, req, future,
249                                                                      this::convertResponse,
250                                                                      this::convertException));
251             return HttpResponse.from(future);
252         }
253 
254         return convertResponse(ctx, result);
255     }
256 
257     private HttpResponse convertResponse(ServiceRequestContext ctx, @Nullable Object result) {
258         try (SafeCloseable ignored = ctx.push(false)) {
259             for (final ResponseConverterFunction func : responseConverters) {
260                 try {
261                     return func.convertResponse(ctx, result);
262                 } catch (FallthroughException ignore) {
263                     // Do nothing.
264                 } catch (Exception e) {
265                     throw new IllegalStateException(
266                             "Response converter " + func.getClass().getName() +
267                             " cannot convert a result to HttpResponse: " + result, e);
268                 }
269             }
270         }
271         // There is no response converter which is able to convert 'null' result to a response.
272         // In this case, we deal the result as '204 No Content' instead of '500 Internal Server Error'.
273         if (result == null) {
274             return HttpResponse.of(HttpStatus.NO_CONTENT);
275         }
276         throw new IllegalStateException(
277                 "No response converter exists for a result: " + result.getClass().getSimpleName());
278     }

280     /**
281      * Returns an {@link HttpResponse} which is created by {@link ExceptionHandlerFunction}.
282      */
283     private HttpResponse convertException(RequestContext ctx, HttpRequest req, Throwable cause) {
284         final Throwable peeledCause = Exceptions.peel(cause);
285 
286         if (Flags.annotatedServiceExceptionVerbosity() == ExceptionVerbosity.ALL &&
287             logger.isWarnEnabled()) {
288             logger.warn("{} Exception raised by method '{}' in '{}':",
289                         ctx, method.getName(), object.getClass().getSimpleName(), peeledCause);
290         }
291 
292         for (final ExceptionHandlerFunction func : exceptionHandlers) {
293             try {
294                 final HttpResponse response = func.handleException(ctx, req, peeledCause);
295                 // Check the return value just in case, then pass this exception to the default handler
296                 // if it is null.
297                 if (response == null) {
298                     break;
299                 }
300                 return response;
301             } catch (FallthroughException ignore) {
302                 // Do nothing.
303             } catch (Exception e) {
304                 logger.warn("{} Unexpected exception from an exception handler {}:",
305                             ctx, func.getClass().getName(), e);
306             }
307         }
308 
309         return HttpResponse.of(HttpStatus.INTERNAL_SERVER_ERROR);
310     }

312     /**
313      * Wraps the specified {@code obj} with {@link CompletableFuture} if it is not an instance of
314      * {@link CompletionStage}.
315      */
316     private static CompletionStage<?> toCompletionStage(Object obj) {
317         if (obj instanceof CompletionStage) {
318             return (CompletionStage<?>) obj;
319         }
320         return CompletableFuture.completedFuture(obj);
321     }

332     /**
333      * Intercepts an {@link HttpResponse} and wraps the response with an {@link ExceptionFilteredHttpResponse}
334      * if it is not an instance of {@link ExceptionFilteredHttpResponse}. This decorator will make an
335      * {@link Exception} to be handled by {@link ExceptionHandlerFunction}s even if the exception is raised
336      * from a decorator.
337      */
338     private class ExceptionFilteredHttpResponseDecorator
339             extends SimpleDecoratingService<HttpRequest, HttpResponse> {
340 
341         ExceptionFilteredHttpResponseDecorator(Service<HttpRequest, HttpResponse> delegate) {
342             super(delegate);
343         }
344 
345         @Override
346         public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) throws Exception {
347             try {
348                 final HttpResponse response = delegate().serve(ctx, req);
349                 if (response instanceof ExceptionFilteredHttpResponse) {
350                     return response;
351                 }
352                 return new ExceptionFilteredHttpResponse(ctx, req, response);
353             } catch (Exception cause) {
354                 return convertException(ctx, req, cause);
355             }
356         }
357     }
358 
359     /**
360      * Intercepts a {@link Throwable} raised from {@link HttpResponse} and then rewrites it as an
361      * {@link HttpResponseException} by {@link ExceptionHandlerFunction}.
362      */
363     private class ExceptionFilteredHttpResponse extends FilteredHttpResponse {
364 
365         private final ServiceRequestContext ctx;
366         private final HttpRequest req;
367 
368         ExceptionFilteredHttpResponse(ServiceRequestContext ctx, HttpRequest req,
369                                       HttpResponse delegate) {
370             super(delegate);
371             this.ctx = ctx;
372             this.req = req;
373         }
374 
375         @Override
376         protected HttpObject filter(HttpObject obj) {
377             return obj;
378         }
379 
380         @Override
381         protected Throwable beforeError(Subscriber<? super HttpObject> subscriber, Throwable cause) {
382             if (cause instanceof HttpResponseException) {
383                 // Do not convert again if it has been already converted.
384                 return cause;
385             }
386             return HttpResponseException.of(convertException(ctx, req, cause));
387         }
388     }
